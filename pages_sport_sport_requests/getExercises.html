Future<List<Exercise>> getExercises(BuildContext context) async {
  final response = await HttpRequest.mainGet('/sports-exercise/', context,
      accessToken: Provider.of<UserProvider>(context, listen: false)
          .getTokens()['access']);
  if (response.statusCode == 200) {
    var body = jsonDecode(response.body);
    return List<Exercise>.from(body.map((e) => Exercise.fromJSON(e)));
  } else {
    throw Exception('Failed to get exercises');
  }
}
