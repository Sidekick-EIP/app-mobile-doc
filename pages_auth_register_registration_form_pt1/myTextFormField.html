Widget myTextFormField(Map<String, dynamic> formDatas,
    {String titleText = "",
    String inputKey = "",
    int maxLength = 0,
    required Map<String, dynamic> validity,
    String suffix = ""}) {
  return Column(
    crossAxisAlignment: CrossAxisAlignment.start,
    children: [
      Padding(
        padding: const EdgeInsets.fromLTRB(0, 0, 0, 16),
        child: Text(titleText, style: const TextStyle(fontSize: 26)),
      ),
      Input(
        key: Key(inputKey),
        maxLength: maxLength == 0 ? TextField.noMaxLength : maxLength,
        showNbChar: maxLength == 0 ? false : true,
        suffix: Text(suffix),
        validator: (value) {
          var res = isFieldValid(validity['regex'].hasMatch(value!), value,
              validity['field_name'], validity['error_message']);
          if (res != null) {
            return res;
          }
          if (validity['condition'] == true &&
              (int.parse(value) < validity['min_value'] ||
                  int.parse(value) > validity['max_value'])) {
            return validity['condition_message'];
          }
          return null;
        },
        onSaved: (value) {
          formDatas[validity['field_name']] = value;
        },
      ),
      const Padding(
        padding: EdgeInsets.fromLTRB(0, 20, 0, 0),
      ),
    ],
  );
}
