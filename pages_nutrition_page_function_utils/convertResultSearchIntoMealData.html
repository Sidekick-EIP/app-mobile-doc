MealData convertResultSearchIntoMealData(
    List<ResultSearch> convertIngredientData, String nameOfTheMeal, int id) {
  List<IngredientData> ingredient = [];
  double totalCalories = 0;

  for (var i in convertIngredientData) {
    totalCalories += i.kcalories;
    ingredient.add(IngredientData(
        name: i.name,
        quantity: i.quantity,
        calories: i.kcalories,
        lipides: i.lipides,
        proteines: i.proteines,
        glucides: i.carbohydrates));
  }
  MealData myMeal = MealData(getIt<MealEditorBlock>().period, id, nameOfTheMeal,
      getIt<MealEditorBlock>().selectedDate, ingredient);
  myMeal.totalCalories = totalCalories;
  return myMeal;
}
