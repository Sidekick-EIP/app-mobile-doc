Future<List<ResultSearch>> getMeal(
    String query, Map<String, String> filter) async {
  Map<String, String> filterParameter = {
    "action": "process",
    "search_terms": query,
    "sort_by": "unique_scans_n",
    "page_size": "30",
    "json": "1"
  };

  filterParameter.addAll(filter);
  var response = await getIt<ApiClient>().get(
    endpoint: "search.pl",
    queryParams: filterParameter,
  );
  if (response.statusCode == 200) {
    var products = response.body["products"] as List;
    if (products.isNotEmpty) {
      var list = products.map((e) => ResultSearch.fromJSON(e)).toList();

      return list;
    } else {
      return [];
    }
  } else {
    throw ("can't fetch search ingredient");
  }
}
