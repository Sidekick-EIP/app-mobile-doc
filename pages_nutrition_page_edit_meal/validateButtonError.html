bool validateButtonError(String value, List<IngredientData> alimentList) {
  final alphanumeric = RegExp(r'^[a-zA-Z0-9\s]+$');
  if (value.isEmpty) {
    return true;
  } else if (value.length < 3) {
    return true;
  } else if (!alphanumeric.hasMatch(value)) {
    return true;
  } else if (alimentList.isEmpty) {
    return true;
  }
  return false;
}
