Future<void> sendUserInfoProfile(
    Map<String, dynamic> userInfos, BuildContext context) async {
  try {
    var accessToken =
        Provider.of<UserProvider>(context, listen: false).getTokens()['access'];
    final response = await HttpRequest.mainPut(
        '/user_infos/update/', userInfos, context,
        accessToken: accessToken);
    if (response.statusCode == 200) {
      Provider.of<UserProvider>(context, listen: false)
          .updateUser(jsonDecode(response.body));
      Navigator.pop(context);
    } else {
      final mySb = MySnackBar(
        text: 'Erreur serveur, vérifiez votre connexion internet.',
        backgroundColor: Colors.red,
      );
      ScaffoldMessenger.of(context).showSnackBar(mySb);
      if (kDebugMode) {
        print("Failed to send new info to server.");
      }
    }
  } catch (e) {
    final mySb = MySnackBar(
      text: 'Erreur serveur, vérifiez votre connexion internet.',
      backgroundColor: Colors.red,
    );
    ScaffoldMessenger.of(context).showSnackBar(mySb);
  }
}
