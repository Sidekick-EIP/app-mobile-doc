String getDropdownValueSportFrequency(value) {
  if (value == "NEVER") {
    return ("Jamais");
  } else if (value == "LESS_THAN_ONCE_A_MONTH") {
    return ("Moins d'une fois par mois");
  } else if (value == "ONCE_A_MONTH") {
    return ("Une fois par mois");
  } else if (value == "ONCE_IN_TWO_WEEKS") {
    return ("Une fois toutes les deux semaines");
  } else if (value == "ONCE_A_WEEK") {
    return ("Une fois par semaine");
  } else if (value == "TWICE_A_WEEK") {
    return ("Deux fois par semaine");
  } else if (value == "THREE_A_WEEK") {
    return ("Trois fois par semaine");
  } else if (value == "FOUR_A_WEEK") {
    return ("Quatre fois par semaine");
  } else if (value == "FIVE_A_WEEK") {
    return ("Cinq fois par semaine");
  } else {
    return "Plus de cinq fois par semaine";
  }
}
